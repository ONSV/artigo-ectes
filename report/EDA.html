na_estado_civil <- sim |>
  filter(is.na(ESTCIV)) |>
  mutate(ANO = year(DTOBITO)) |>
  count(ANO)

na_escolaridade <- sim |>
  filter(is.na(ESC_DECOD)) |>
  mutate(ANO = year(DTOBITO)) |>
  count(ANO)

na_ambos <- sim |>
  filter(is.na(ESTCIV) & is.na(ESC_DECOD)) |>
  mutate(ANO = year(DTOBITO)) |>
  count(ANO)

na_total <- sim |>
  mutate(ANO = year(DTOBITO)) |>
  count(ANO)

na_df <- reduce(
  list(
    na_estado_civil,
    na_escolaridade,
    na_ambos,
    na_total
  ),
  left_join,
  by = "ANO"
) |>
  rename(
    estado_civil = n.x,
    escolaridade = n.y,
    ambos = n.x.x,
    total = n.y.y
  )

na_df |>
  pivot_longer(-1) |>
  ggplot(aes(x = ANO, y = value, color = name)) +
  geom_point() +
  geom_line()

na_df |>
  mutate(
    taxa_estado_civil = estado_civil / total,
    taxa_escolaridade = escolaridade / total,
    taxa_ambos = ambos / total
  ) |>
  select(ANO, taxa_estado_civil, taxa_escolaridade, taxa_ambos) |>
  pivot_longer(-1) |>
  ggplot(aes(x = ANO, y = value, color = name)) +
